[docs]def is_valid_address(address):
    """Determines if the provided string is a valid Stellar address.

    :param str address: address to check

    :return: True if this is a correct address
    :rtype: boolean
    """
    if len(address) != 56:
        return False

    try:
        decode_check('account', address)
        return True
    except:
        return False
